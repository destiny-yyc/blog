function FlatToTree(arr) {
  const map = arr.reduce((acc, val) => {
    acc[val.id] = val
    return acc
  }, {})
  const tree = []
  arr.forEach(region => {
    if (region.parent) {
      const parent = map[region.parent]
      if (!parent.children) {
        parent.children = [region]
      }
      else {
        parent.children.push(region)
      }
    }
    else {
      tree.push(region)
    }
  })
  return { tree }
}
